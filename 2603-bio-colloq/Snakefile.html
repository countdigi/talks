rule hds:
    """
    START OF PIPELINE
    - We remove '##filedate' in the header before running HDS_UTIL as if the header is there it
      hds-util will update it therefore creating a different checksum each time which makes testing
      changes to this rule annoying.
    - If TEST_REGIONS is set, we create a much smaller output file for quick iterative testing.
    """
    input:
        vcf=lambda wc: join(INPUT_COMBINED[wc.chip], "{chr}.vcf.gz"),
    output:
        vcf=wrk("hds/{chip}/{chr}.vcf.gz"),
    resources:
        mem="20G",
        runtime="21d",
    threads:
        2
    benchmark:
        mtr("hds/{chip}/{chr}.txt")
    shell:
        """
        _bcftools_opts=""

        if [[ -n '{TEST_REGIONS}' ]]; then
          _bcftools_opts="--regions-file={TEST_REGIONS}"
        fi

        {{
          {BCFTOOLS} view --no-version --header-only {input.vcf} | /bin/sed '/^##filedate/d'
          {BCFTOOLS} view --no-version --no-header $_bcftools_opts {input.vcf}
        }} \
        | {BCFTOOLS} annotate --no-version --remove='INFO/AF,INFO/MAF' \
        | {BCFTOOLS} plugin fill-tags --no-version -- --tags='INFO/AC,INFO/AF,INFO/AN,INFO/MAF' \
        \
        | {HDS_UTIL} --format=GT,GP,DS,HDS,SD --output-format=vcf.gz --output={output.vcf}

        {BCFTOOLS} index --tbi {output.vcf}

        for f in {output.vcf} {output.vcf}.tbi; do
          /bin/md5sum $f | cut -d' ' -f1 > $f.md5
        done
        """

rule filter_r2:
    """
    NOTE: To avoid confusion we strip r2Filter= in header since we filter w/ our own R2_VALUE.
    """
    input:
        vcf=wrk("hds/{chip}/{chr}.vcf.gz"),
    output:
        vcf=wrk("filter_r2/{chip}/{chr}.vcf.gz"),
    params:
        r2=lambda wc: R2_VALUE[wc.chip],
    resources:
        mem="10G",
        runtime="7d",
    threads:
        2
    benchmark:
        mtr("filter_r2/{chip}/{chr}.txt")
    shell:
        """
        _header() {{ {BCFTOOLS} view --no-version --header-only {input.vcf} | /bin/egrep -v '^##r2Filter='; }}

        {BCFTOOLS} reheader --header <(_header) {input.vcf} \
          | {BCFTOOLS} view --no-version --include 'R2>{params.r2}' --output={output.vcf}

        {BCFTOOLS} index --tbi {output.vcf}

        for f in {output.vcf} {output.vcf}.tbi; do
          /bin/md5sum $f | cut -d' ' -f1 > $f.md5
        done
        """

rule plink:
    """
    We use plink option --output-chr 26 to convert X to 23, needed for compatibility with king and xwas

    For wildcards.ethnicity == 'eur':
      Keep only non-related, European controls.
      The keep_file only contains the IID so we need to use this to create the 2-column FID/IID keep file required by plink.
    """
    input:
        vcf=wrk("filter_r2/{chip}/{chr}.vcf.gz"),
        fams_orig=FAM_ORIG.values(),
        eur_controls=EUR_CONTROLS,
    output:
        pfx=wrk("plink/{chip}/{ethnicity}/{chr}"),
    shadow:
        "shallow"
    resources:
        mem="60G",
        runtime="4h",
    threads:
        4
    benchmark:
        mtr("plink/{chip}/{ethnicity}/{chr}.txt")
    shell:
        """
        _opts_plink=""

        if [[ '{wildcards.ethnicity}' == 'eur' ]]; then
          {BCFTOOLS} query --list-samples {input.vcf} | tr '_' '\t' > samples.tsv

          set +x
          for iid in $(tail -n +2 {input.eur_controls} | cut -f1); do
            awk --assign iid=$iid '{{ if ($2 == iid) print $1 "\t" $2 }}' samples.tsv
          done > eur_controls.tsv
          set -x

          _opts_plink="--keep eur_controls.tsv"
        fi

        {PLINK20} \
          $_opts_plink \
          --silent \
          --vcf {input.vcf} 'dosage=HDS' \
          --make-pgen psam-cols=fid,parents,sex,pheno1 \
          --id-delim _ \
          --output-chr 26 \
          --out out

        /bin/mv out.pvar {output.pfx}.pvar
        /bin/mv out.pgen {output.pfx}.pgen
        {PYTHON} {BASE}/scripts/update-fam.py --type=psam out.psam {input.fams_orig} > {output.pfx}.psam

        /bin/touch {output.pfx}
        """

def inp_keep_pfx(wc):
    if wc.chr == "chr6": input_dir = "hwe_hla"
    elif wc.chr == "chrX": input_dir = "hwe_sex"
    else: input_dir = "hwe"

    return wrk(input_dir, wc.chip, "eur", wc.sex, wc.chr)

rule mendel:
    """
    Filter SNPs using mendelian error checking in plink. We use wildcards {me} which is fed to the plink command "--me"
    (Filter out trios and variants with high Mendel error rates) to allow us to experiment with different thresholds.

    NOTE: For the downstream rule we use the single value set in ME_THRESHOLD.

    We also determined that we want to exclude immunochip from the mendelian error analysis so we exclude all
    immunochip samples at the beginning of this rule using the hds_util/immun rule to filter immun subjects (n=5983) out.
    """
    input:
        pfx=wrk("combine_plink/{sex}/{chr}"),
        vcf=wrk("filter_r2/immun/{chr}.vcf.gz"),
    output:
        pfx=wrk("mendel/{me}/{sex}/{chr}"),

    shell:
        """

        {PLINK19} --silent --make-bed --bfile {input.pfx} --remove immun_subjects.txt --out no_immun

        # Run statistics

        {PLINK19} --silent --bfile no_immun --mendel --out out.mendel
        {PLINK19} --silent --bfile no_immun --mendel --mendel-duos --out out.mendel.mendel_duos

        {PLINK19} --silent --make-bed --bfile no_immun --me 1 {wildcards.me} 'var-first' --mendel-duos --set-me-missing \
                  --out out.filter.1

        {PLINK19} --silent --make-bed --bfile out.filter.1 --set-hh-missing --out out.filter.2

        {PLINK19} --silent --make-bed --bfile out.filter.2 --geno 1 --out out.filter.3

        done

        /bin/touch {output.pfx}
        """

rule cc_samples_common:
    """
    Find the samples common between WGS and GWAS samples then generate mapping files with their specific sample name format
    to a common sample name format.
    """
    input:
        wgs_samples=wrk("cc_samples/wgs/{chr}.txt"),
        gwas_samples=wrk("cc_samples/gwas/{chr}.txt")
    output:
        wgs_map=wrk("cc_samples_common/wgs/{chr}.tsv"),
        gwas_map=wrk("cc_samples_common/gwas/{chr}.tsv")
    resources:
        mem="10G",
        runtime="1h",
    threads:
        1
    run:
        db_wgs = {}
        db_gwas = {}

        with open(input.wgs_samples) as f:
            for line in f:
                iid, phase = line.rstrip("\n").split("_")
                if iid in db_wgs:
                    if phase == "P2":
                        db_wgs[iid] = iid + "_" + phase
                else:
                    db_wgs[iid] = iid + "_" + phase

        with open(input.gwas_samples) as f:
            for line in f:
                fid, iid = line.rstrip("\n").split("_")
                if iid in db_gwas:
                    print(f"Error {iid} duplicate in db_gwas. Exiting.", file=sys.stderr)
                    sys.exit(1)
                else:
                    db_gwas[iid] = fid + "_" + iid

        samples_common = set(db_wgs).intersection(db_gwas)

        with open(output.wgs_map, "w") as f_out:
            for sample in samples_common:
                print(" ".join([db_wgs[sample], sample]), file=f_out)

        with open(output.gwas_map, "w") as f_out:
            for sample in samples_common:
                print(" ".join([db_gwas[sample], sample]), file=f_out)

rule cc_filter_ruth_snps:
    """
    Generate a list of SNPs to exclude. Downstream we only use the 'controls' wildcard output as the
    filter list to apply to the full cc_filter/all but we generate an 'all' wildcard version for comparison
    purposes only.

    The statistic of interest is HWE_SLP_I, which is the signed log10 P-value of the HWE test with individual-specific
    allele frequencies, adjusted for population structure:
    - HWE_SLP_I < 0 indicates an excess of heterozygotes
    - HWE_SLP_I > 0 indicates heterozygote depletion

    An excess of heterozygotes can be a telltale sign of certain types of technical artefacts
    """
    input:
        pca_tsv=wrk("cc_filter_ruth_pca/{subset}/{sex}/{chr}.tsv"),
        vcf=wrk("cc_filter/{subset}/{sex}/{chr}.vcf.gz")
    output:
        snps_exclude=wrk("cc_filter_ruth_snps/{subset}/{sex}/{chr}.txt")
    params:
        filters=params_cc_filter_ruth_snps
    shadow:
        "shallow"
    resources:
        mem="30G",
        runtime="1d",
    threads:
        2
    benchmark:
        mtr("cc_filter_ruth_snps/{subset}/{sex}/{chr}.txt")
    shell:
        """
        if [[ '{wildcards.chr}' == 'chrX' ]]; then _ruth_opts='--x-label chrX'; else _ruth_opts=''; fi

        {RUTH} \
          --vcf {input.vcf} \
          --evec {input.pca_tsv} \
          --field GT \
          --lambda 0 \
          --lrt-em \
          $_ruth_opts \
          --out ruth.vcf

        {BCFTOOLS} view --output=ruth.bcf ruth.vcf

        {BCFTOOLS} index ruth.bcf

        for _filter in {params.filters}; do
            _hwe_slp_i_max=$(echo $_filter | cut -d, -f1)
            _range=$(echo $_filter | cut -d, -f2)

            {BCFTOOLS} view ruth.bcf $_range \
              | {BCFTOOLS} query --include="HWE_SLP_I > $_hwe_slp_i_max" --format='%ID\t%POS\t%REF\t%ALT\n'

        done | sort --version-sort > {output.snps_exclude}.full

        cut -f1 {output.snps_exclude}.full | sort --version-sort | uniq > {output.snps_exclude}
        """
